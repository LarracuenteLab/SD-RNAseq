# SD-RNAseq
# This  repository contains files and code to recreate the analysis and figures in Edvalson, Wei, et al. 2025
# File list is below:

Rsp_SD_Code.Rmd = Contains all functions and packages needed to run the other R markdown files **RUN FIRST**

Rsp_SD_smRNA.rmd = Contains the code that runs differential expression on smRNA

Rsp_SD_totRNA.rmd = Contains the code that runs differential expression on totRNA

Rsp_SD_Figures.rmd = Contains the code for generating figure panels that are not generated by the smRNA or totRNA markdown

TE_Families.tsv = table defining what family each repeat annotation is a member of

miRNA.list = List of micro RNA FBgn IDs from Flybase

PANGEA_Results = Directory containing the csvs for all PANGEA analyses
    -SD5vR16_Gla_PANGEA.csv = PANGEA Results for SD-5/Gla compared to R16/Gla
    -SDMadvR16_Gla_PANGEA.csv = PANGEA Results for SD-Mad/Gla compared to R16/Gla
    -Module_A_PANGEA.csv = PANGEA Results for Module A from WGCNA analysis
    -Module_B_PANGEA.csv = PANGEA Results for Module B from WGCNA analysis
    -Module_C_PANGEA.csv = PANGEA Results for Module C from WGCNA analysis
    -Module_F_PANGEA.csv = PANGEA Results for Module F from WGCNA analysis
    -Module_G_PANGEA.csv = PANGEA Results for Module G from WGCNA analysis
    -Module_H_PANGEA.csv = PANGEA Results for Module H from WGCNA analysis

smRNA_counts = contains all counts for small RNA seq
    all_summedbyFam = subdirectory containing the counts with annotations summed by repeat family
        
totRNA_Counts = Contains all total RNA counts
    HTSeq_Genes = subdirectory containing the counts for genes as generated by HTSeq 
    PropCount_Repeats = subdirectory containing the counts for repeat families 

supplementalTables = Destination directory for all tables generated from smRNA and totRNA analysis
